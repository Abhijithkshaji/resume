document.getElementById("resumeForm").addEventListener("submit", function(event) { event.preventDefault(); const fileInput = document.getElementById("resumeFile"); const file = fileInput.files[0]; if (file) { const formData = new FormData(); formData.append("resume", file); fetch("http://127.0.0.1:8000/predict/", { // Django backend URL method: "POST", body: formData }) .then(response => response.json()) .then(data => { if (data.prediction) { document.getElementById("result").innerText = `Predicted Category: ${data.prediction}`; } else { document.getElementById("result").innerText = "Error: Unable to classify the resume."; } }) .catch(error => { document.getElementById("result").innerText = "An error occurred."; }); } else { document.getElementById("result").innerText = "Please upload a valid resume file."; } });
